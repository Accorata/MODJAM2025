	portrait = "GFX_crisis_icon_fe_ship_1"

	triggered_modifier = {
		potential = {
			has_ancrel = yes
		}
		ship_archaeological_site_clues_add = 1
	}

	triggered_modifier = {
		potential = {
			has_ancrel = no
		}
		scientist_exp_gain = 0.20
	}

	on_unlock = {
		if = {
			limit = {
				has_ancrel = yes
			}
			custom_tooltip = unlocks_otc_secrets_of_the_past
		}
		else = {
			custom_tooltip = unlocks_otc_secrets_of_the_past_no_ancrel
		}
	}
}
